Overview:

This was one of the first practice exercises I did for Cypress. The webpage itself is meant for using Test Cafe, however it works just as well with Cypress.

Test Execution:

My Cypress Code:

    
      describe('TestCafe Elements Spec', function() {
        it('Interacts with various elements on the TestCafe Example page', function() {
            cy.visit('https://devexpress.github.io/testcafe/example/')
           
            //Gets the developer name input and types my name
            cy.get('#developer-name').type('Adrian')
    
            //Gets and clicks the radio button for Windows
            cy.get(':nth-child(1) > :nth-child(2) > label').click()
    
            //Gets and clicks the parallel testing checkbox
            cy.get('#background-parallel-testing').click()
    
            //Gets and clicks the continuous integration checkbox
            cy.get('#continuous-integration-embedding').click()
    
            //Chooses a value from the dropdown box
            cy.get('#preferred-interface').select('Both')
    
            //Checks that the slider is disabled on the UI
            cy.get('div[id="slider"]').should('have.class', 'ui-state-disabled')
    
            //Clicks the Test Cafe checkbox
            cy.get('#tried-test-cafe').click()
    
            //Checks that the slider is now enabled on the UI
            cy.get('div[id="slider"]').should('not.have.class', 'ui-state-disabled')
    
            //Gets and clicks the slider handle and drags it to the 6th position
            cy.get('.ui-slider-handle').trigger('mousedown', { which: 1 }) .trigger('mousemove', { clientX: 400, clientY: 500 })
    
            //Checks that the comment box is empty
            cy.get('#comments').should('not.have.value')
    
            //Types a message in the comment box
            cy.get('#comments').click().type('Test Cafe is alright.')
    
            //Checks that the comment box contains the message
            cy.get('#comments').should('have.value', "Test Cafe is alright.")
    
            //Clears the comment box
            cy.get('#comments').click().clear()
    
            //Checks that the comment box is empty again
            cy.get('#comments').should('not.have.value')
    
            //Clicks the submit button
            cy.get('#submit-button').click()
    
            //After the new page loads, checks that header contains my name
            cy.get('#article-header').should('contain', 'Adrian')
    
            //Checks that the new page contains this link
            cy.get('p').should('contain', 'devexpress.github.io/testcafe')
    
        })
    })
